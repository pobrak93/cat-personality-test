const questions = [
    { 
        question: "คุณชอบทำอะไรเวลาว่าง?", 
        choices: ["นอนเล่น", "สำรวจสิ่งใหม่ๆ", "อยู่เงียบๆ คนเดียว", "เข้าสังคม"]
    },
    { 
        question: "คุณคิดว่าตัวเองเป็นคนแบบไหน?", 
        choices: ["ขี้อ้อน", "ฉลาด", "สุขุม", "ร่าเริง"]
    }
];

let currentQuestion = 0;
let answers = [];

function showQuestion() {
    const questionElem = document.getElementById("question");
    const choicesElem = document.getElementById("choices");
    
    questionElem.innerText = questions[currentQuestion].question;
    choicesElem.innerHTML = "";

    questions[currentQuestion].choices.forEach((choice, index) => {
        const button = document.createElement("button");
        button.innerText = choice;
        button.classList.add("btn");
        button.onclick = () => {
            answers.push(choice);
            nextQuestion();
        };
        choicesElem.appendChild(button);
    });
}

function nextQuestion() {
    if (currentQuestion < questions.length - 1) {
        currentQuestion++;
        showQuestion();
    } else {
        calculateResult();
    }
}

function calculateResult() {
    let result = "Scottish Fold"; // ค่าเริ่มต้น
    if (answers.includes("ฉลาด")) {
        result = "Bengal";
    } else if (answers.includes("สุขุม")) {
        result = "British Shorthair";
    }
    
    localStorage.setItem("catResult", result);
    window.location.href = "result.html";
}

document.addEventListener("DOMContentLoaded", showQuestion);
